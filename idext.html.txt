        // ========== UTILITY FUNCTIONS ==========
        function generateStars(rating) {
            let stars = '';
            for (let i = 1; i <= 5; i++) {
                if (i <= Math.floor(rating)) stars += '<i class="fas fa-star"></i>';
                else if (i === Math.ceil(rating) && !Number.isInteger(rating)) stars += '<i class="fas fa-star-half-alt"></i>';
                else stars += '<i class="far fa-star"></i>';
            }
            return stars;
        }

        function showToast(message, type = 'success') {
            const container = document.getElementById('toastContainer');
            const toast = document.createElement('div');
            toast.className = `toast ${type}`;
            const icon = type === 'success' ? 'fa-check-circle' : type === 'error' ? 'fa-exclamation-circle' : 'fa-exclamation-triangle';
            toast.innerHTML = `<i class="fas ${icon} text-${type === 'success' ? 'success' : type === 'error' ? 'danger' : 'warning'}"></i><span>${message}</span>`;
            container.appendChild(toast);
            setTimeout(() => toast.remove(), 4000);
        }

        // ========== NAVIGATION ==========
        function showPage(pageName) {
            document.querySelectorAll('.page-section').forEach(s => s.classList.remove('active'));
            document.querySelectorAll('.nav-link').forEach(l => l.classList.remove('active'));

            const target = document.getElementById('page-' + pageName);
            if (target) {
                target.classList.add('active');
                window.scrollTo({ top: 0, behavior: 'smooth' });
            }

            // Update nav active state
            const navLinks = document.querySelectorAll('.nav-link');
            navLinks.forEach(link => {
                if (link.getAttribute('onclick') && link.getAttribute('onclick').includes("'" + pageName + "'")) {
                    link.classList.add('active');
                }
            });

            // Close mobile menu
            const navbarCollapse = document.querySelector('.navbar-collapse');
            if (navbarCollapse && navbarCollapse.classList.contains('show')) {
                navbarCollapse.classList.remove('show');
            }

            // Load page-specific content
            if (pageName === 'home') renderHomePage();
            if (pageName === 'blog') renderBlogPage();
            if (pageName === 'products') renderProductsPage();
            if (pageName === 'comparisons') renderComparisonsPage();
        }

        // ========== SEARCH ==========
        function handleSearch(query) {
            const resultsDiv = document.getElementById('searchResults');
            if (!query || query.length < 2) {
                resultsDiv.classList.remove('show');
                return;
            }

            const lowerQuery = query.toLowerCase();
            const productResults = products.filter(p => p.name.toLowerCase().includes(lowerQuery));
            const blogResults = blogPosts.filter(b => b.title.toLowerCase().includes(lowerQuery));

            if (productResults.length === 0 && blogResults.length === 0) {
                resultsDiv.innerHTML = '<div class="search-result-item"><div class="text-muted">No results found</div></div>';
            } else {
                resultsDiv.innerHTML = '';
                if (productResults.length > 0) {
                    resultsDiv.innerHTML += '<div class="px-3 py-2 fw-bold text-muted" style="font-size:0.8rem;text-transform:uppercase;">Products</div>';
                    resultsDiv.innerHTML += productResults.slice(0, 3).map(p => `
                        <div class="search-result-item" onclick="showProductDetail(${p.id}); document.getElementById('searchResults').classList.remove('show');">
                            <img src="${p.image}" alt="${p.name}">
                            <div><div class="fw-bold">${p.name}</div><div style="font-size:0.8rem;color:var(--text-gray);">${p.price}</div></div>
                        </div>
                    `).join('');
                }
                if (blogResults.length > 0) {
                    resultsDiv.innerHTML += '<div class="px-3 py-2 fw-bold text-muted" style="font-size:0.8rem;text-transform:uppercase;">Articles</div>';
                    resultsDiv.innerHTML += blogResults.slice(0, 3).map(b => `
                        <div class="search-result-item" onclick="showBlogDetail(${b.id}); document.getElementById('searchResults').classList.remove('show');">
                            <img src="${b.image}" alt="${b.title}">
                            <div><div class="fw-bold">${b.title}</div><div style="font-size:0.8rem;color:var(--text-gray);">by ${b.author}</div></div>
                        </div>
                    `).join('');
                }
            }
            resultsDiv.classList.add('show');
        }

        // ========== SCROLL EFFECTS ==========
        function scrollToTop() {
            window.scrollTo({ top: 0, behavior: 'smooth' });
        }

        window.addEventListener('scroll', () => {
            const navbar = document.getElementById('navbar');
            const backToTop = document.getElementById('backToTop');

            if (window.scrollY > 100) {
                navbar.classList.add('sticky');
                backToTop.classList.add('show');
            } else {
                navbar.classList.remove('sticky');
                backToTop.classList.remove('show');
            }

            // Fade in animations
            document.querySelectorAll('.fade-in').forEach(el => {
                const rect = el.getBoundingClientRect();
                if (rect.top < window.innerHeight - 100) {
                    el.classList.add('visible');
                }
            });
        });

        // Close search results when clicking outside
        document.addEventListener('click', (e) => {
            const searchBox = document.querySelector('.search-box');
            if (searchBox && !searchBox.contains(e.target)) {
                document.getElementById('searchResults').classList.remove('show');
            }
        });
